Modal.razor 

@* Modal.razor *@
@if (IsVisible)
{
    <div class="modal" style="display: flex;" @onclick="OnBackdropClick">
        <div class="modal-content" @onclick:stopPropagation="true">
            <span class="close" @onclick="Close">×</span>
            <h3>@Title</h3>
            <div>@ChildContent</div>
            @if (ShowActions)
            {
                <div class="modal-actions">
                    <button class="btn btn-secondary" @onclick="Close">@CancelText</button>
                    <button class="btn btn-primary" @onclick="OnConfirm">@ConfirmText</button>
                </div>
            }
        </div>
    </div>
}

@code {
    [Parameter] public string Title { get; set; } = "Modal";
    [Parameter] public bool IsVisible { get; set; }
    [Parameter] public EventCallback OnClose { get; set; }
    [Parameter] public EventCallback OnConfirmClick { get; set; }
    [Parameter] public RenderFragment? ChildContent { get; set; }
    [Parameter] public bool ShowActions { get; set; } = true;
    [Parameter] public string CancelText { get; set; } = "Cancelar";
    [Parameter] public string ConfirmText { get; set; } = "Confirmar";

    private async Task Close()
    {
        IsVisible = false;
        await OnClose.InvokeAsync();
    }

    private async Task OnConfirm()
    {
        await OnConfirmClick.InvokeAsync();
        await Close();
    }

    private async Task OnBackdropClick()
    {
        await Close();
    }
}

Uso: <Modal @bind-IsVisible="showModal" Title="Meu Modal">Conteúdo</Modal>

ToastService.cs + Toast.razor 

// ToastService.cs
public class ToastService
{
    public event Action<string, ToastType, string>? OnShow;
    
    public void Show(string message, ToastType type = ToastType.Info, 
                     string position = "top-right")
    {
        OnShow?.Invoke(message, type, position);
    }

    public void Success(string message) => Show(message, ToastType.Success);
    public void Error(string message) => Show(message, ToastType.Error);
    public void Warning(string message) => Show(message, ToastType.Warning);
    public void Info(string message) => Show(message, ToastType.Info);
}

public enum ToastType { Success, Error, Warning, Info }

@* Toast.razor *@
@inject ToastService ToastService
@implements IDisposable

@if (_isVisible)
{
    <div class="toast @_position show" style="background-color: @_bgColor">
        @_message
    </div>
}

@code {
    private bool _isVisible;
    private string _message = "";
    private string _position = "top-right";
    private string _bgColor = "";

    protected override void OnInitialized()
    {
        ToastService.OnShow += ShowToast;
    }

    private async void ShowToast(string message, ToastType type, string position)
    {
        _message = message;
        _position = position;
        _bgColor = type switch
        {
            ToastType.Success => "var(--success)",
            ToastType.Error => "var(--error)",
            ToastType.Warning => "var(--warning)",
            _ => "var(--info)"
        };
        
        _isVisible = true;
        StateHasChanged();
        
        await Task.Delay(3000);
        _isVisible = false;
        StateHasChanged();
    }

    public void Dispose() => ToastService.OnShow -= ShowToast;
}

Registrar: builder.Services.AddScoped<ToastService>();
Uso: @inject ToastService Toast → Toast.Success("Sucesso!");

Alert.razor 

@* Alert.razor *@
@if (IsVisible)
{
    <div class="alert alert-@Type.ToString().ToLower()">
        <i class="bi bi-@GetIcon() alert-icon"></i>
        <div class="alert-content">
            @if (!string.IsNullOrEmpty(Title))
            {
                <strong>@Title</strong>
            }
            @Message
            @ChildContent
        </div>
        @if (Dismissible)
        {
            <i class="bi bi-x alert-close" @onclick="Dismiss"></i>
        }
    </div>
}

@code {
    [Parameter] public AlertType Type { get; set; } = AlertType.Info;
    [Parameter] public string? Title { get; set; }
    [Parameter] public string? Message { get; set; }
    [Parameter] public bool IsVisible { get; set; } = true;
    [Parameter] public bool Dismissible { get; set; } = true;
    [Parameter] public EventCallback OnDismiss { get; set; }
    [Parameter] public RenderFragment? ChildContent { get; set; }

    private async Task Dismiss()
    {
        IsVisible = false;
        await OnDismiss.InvokeAsync();
    }

    private string GetIcon() => Type switch
    {
        AlertType.Success => "check-circle-fill",
        AlertType.Error => "x-circle-fill",
        AlertType.Warning => "exclamation-triangle-fill",
        _ => "info-circle-fill"
    };
}

public enum AlertType { Success, Error, Warning, Info }

Uso: <Alert Type="AlertType.Success" Title="Sucesso!" Message="Operação concluída" />

Dicas de Implementação

  • State Management: Use Cascading Parameters ou Services para gerenciar estados globais
  • Animações: Combine CSS transitions com @key directive para animações suaves
  • Acessibilidade: Sempre inclua aria-labels e role attributes
  • Performance: Use @key para otimizar rerenders em listas
  • Skeleton Loading: Implemente usando @if (!dataLoaded) para melhor UX

Accordion.razor 

@* Accordion.razor *@
<div class="accordion">
    @foreach (var item in Items)
    {
        <div class="accordion-item">
            <button class="accordion-header" @onclick="() => Toggle(item.Id)">
                @item.Title
                <i class="bi bi-chevron-down accordion-icon @(IsOpen(item.Id) ? "open" : "")"></i>
            </button>
            @if (IsOpen(item.Id))
            {
                <div class="accordion-content" style="display: block;">
                    @item.Content
                </div>
            }
        </div>
    }
</div>

@code {
    [Parameter] public List<AccordionItem> Items { get; set; } = new();
    [Parameter] public bool AllowMultiple { get; set; } = false;
    
    private HashSet<string> openItems = new();

    private void Toggle(string id)
    {
        if (openItems.Contains(id))
        {
            openItems.Remove(id);
        }
        else
        {
            if (!AllowMultiple)
                openItems.Clear();
            openItems.Add(id);
        }
    }

    private bool IsOpen(string id) => openItems.Contains(id);
}

public class AccordionItem
{
    public string Id { get; set; } = Guid.NewGuid().ToString();
    public string Title { get; set; } = "";
    public RenderFragment? Content { get; set; }
}

Uso: <Accordion Items="accordionItems" AllowMultiple="false" />

Tabs.razor 

@* Tabs.razor *@
<div class="tabs">
    <div class="tabs-header">
        @foreach (var tab in TabItems)
        {
            <button class="tab-button @(ActiveTab == tab.Id ? "active" : "")" 
                    @onclick="() => SetActiveTab(tab.Id)">
                @tab.Title
            </button>
        }
    </div>
    <div class="tabs-content">
        @foreach (var tab in TabItems)
        {
            <div class="tab-panel @(ActiveTab == tab.Id ? "active" : "")">
                @tab.Content
            </div>
        }
    </div>
</div>

@code {
    [Parameter] public List<TabItem> TabItems { get; set; } = new();
    [Parameter] public string ActiveTab { get; set; } = "";
    [Parameter] public EventCallback<string> OnTabChanged { get; set; }

    protected override void OnInitialized()
    {
        if (string.IsNullOrEmpty(ActiveTab) && TabItems.Any())
            ActiveTab = TabItems.First().Id;
    }

    private async Task SetActiveTab(string tabId)
    {
        ActiveTab = tabId;
        await OnTabChanged.InvokeAsync(tabId);
    }
}

public class TabItem
{

Stepper.razor 

@* Stepper.razor *@
<div class="stepper">
    @for (int i = 0; i < Steps.Count; i++)
    {
        var index = i;
        var step = Steps[i];
        var stepClass = GetStepClass(index);
        
        <div class="step @stepClass">
            <div class="step-circle">
                @if (index < CurrentStep)
                {
                    <i class="bi bi-check"></i>
                }
                else
                {
                    @(index + 1)
                }
            </div>
            <span class="step-label">@step.Label</span>
        </div>
        
        @if (i < Steps.Count - 1)
        {
            <div class="step-line"></div>
        }
    }
</div>

@code {
    [Parameter] public List<StepItem> Steps { get; set; } = new();
    [Parameter] public int CurrentStep { get; set; } = 0;

    private string GetStepClass(int index)
    {
        if (index < CurrentStep) return "completed";
        if (index == CurrentStep) return "active";
        return "";
    }
}

public class StepItem
{
    public string Label { get; set; } = "";
    public RenderFragment? Content { get; set; }
}

Uso: <Stepper Steps="steps" CurrentStep="2" />

ContextMenu.razor 

@* ContextMenu.razor *@
@inject IJSRuntime JS

<div @ref="containerRef" 
     @oncontextmenu="HandleContextMenu" 
     @oncontextmenu:preventDefault="true">
    @ChildContent
</div>

@if (IsVisible)
{
    <div class="context-menu" 
         style="left: @($"{Position.X}px"); top: @($"{Position.Y}px")"
         @onclick:stopPropagation="true">
        @foreach (var item in MenuItems)
        {
            @if (item.IsDivider)
            {
                <div class="context-menu-divider"></div>
            }
            else
            {
                <a class="context-menu-item" 
                   @onclick="() => HandleItemClick(item)">
                    @if (!string.IsNullOrEmpty(item.Icon))
                    {
                        <i class="bi bi-@item.Icon"></i>
                    }
                    @item.Text
                </a>
            }
        }
    </div>
}

@code {
    [Parameter] public List<ContextMenuItem> MenuItems { get; set; } = new();
    [Parameter] public EventCallback<ContextMenuItem> OnItemClick { get; set; }
    [Parameter] public RenderFragment? ChildContent { get; set; }
    
    private ElementReference containerRef;
    private bool IsVisible { get; set; }
    private (int X, int Y) Position { get; set; }

    protected override async Task OnAfterRenderAsync(bool firstRender)
    {
        if (firstRender)
        {
            await JS.InvokeVoidAsync("addEventListener", "click", 
                DotNetObjectReference.Create(this), "HideMenu");
        }
    }

    private void HandleContextMenu(MouseEventArgs e)
    {
        Position = ((int)e.ClientX, (int)e.ClientY);
        IsVisible = true;
        StateHasChanged();
    }

    [JSInvokable]
    public void HideMenu()
    {
        IsVisible = false;
        StateHasChanged();
    }

    private async Task HandleItemClick(ContextMenuItem item)
    {
        await OnItemClick.InvokeAsync(item);
        IsVisible = false;
    }
}

public class ContextMenuItem
{
    public string Text { get; set; } = "";
    public string? Icon { get; set; }
    public bool IsDivider { get; set; }
    public object? Value { get; set; }
}

Pagination.razor 

@* Pagination.razor *@
<div class="pagination">
    <button class="pagination-btn" 
            @onclick="() => ChangePage(1)" 
            disabled="@(CurrentPage == 1)">
        <i class="bi bi-chevron-double-left"></i>
    </button>
    
    <button class="pagination-btn" 
            @onclick="() => ChangePage(CurrentPage - 1)" 
            disabled="@(CurrentPage == 1)">
        <i class="bi bi-chevron-left"></i>
    </button>
    
    @foreach (var page in GetVisiblePages())
    {
        var pageNumber = page;
        <button class="pagination-btn @(CurrentPage == page ? "active" : "")" 
                @onclick="() => ChangePage(pageNumber)">
            @page
        </button>
    }
    
    <button class="pagination-btn" 
            @onclick="() => ChangePage(CurrentPage + 1)" 
            disabled="@(CurrentPage == TotalPages)">
        <i class="bi bi-chevron-right"></i>
    </button>
    
    <button class="pagination-btn" 
            @onclick="() => ChangePage(TotalPages)" 
            disabled="@(CurrentPage == TotalPages)">
        <i class="bi bi-chevron-double-right"></i>
    </button>
</div>

@code {
    [Parameter] public int CurrentPage { get; set; } = 1;
    [Parameter] public int TotalPages { get; set; } = 1;
    [Parameter] public EventCallback<int> OnPageChanged { get; set; }
    [Parameter] public int MaxVisiblePages { get; set; } = 5;

    private async Task ChangePage(int page)
    {
        if (page < 1 || page > TotalPages) return;
        await OnPageChanged.InvokeAsync(page);
    }

    private List<int> GetVisiblePages()
    {
        var pages = new List<int>();
        var start = Math.Max(1, CurrentPage - MaxVisiblePages / 2);
        var end = Math.Min(TotalPages, start + MaxVisiblePages - 1);
        
        if (end - start < MaxVisiblePages - 1)
            start = Math.Max(1, end - MaxVisiblePages + 1);
        
        for (int i = start; i <= end; i++)
            pages.Add(i);
        
        return pages;
    }
}

                    <div class="mega-menu-column">
                        <h4>
                            @if (!string.IsNullOrEmpty(column.Icon))
                            {
                                <i class="bi bi-@column.Icon"></i>
                            }
                            @column.Title
                        </h4>
                        <ul>
                            @foreach (var item in column.Items)
                            {
                                <li>
                                    <a href="@item.Url" @onclick="() => HandleItemClick(item)">
                                        @if (!string.IsNullOrEmpty(item.Icon))
                                        {
                                            <i class="bi bi-@item.Icon"></i>
                                        }
                                        @item.Text
                                        @if (!string.IsNullOrEmpty(item.Description))
                                        {
                                            <small>@item.Description</small>
                                        }
                                    </a>
                                </li>
                            }
                        </ul>
                    </div>
                }
            </div>
        </div>
    }
</div>

    [Parameter] public string ButtonClass { get; set; } = "btn-secondary";
    [Parameter] public List<MegaMenuColumn> MenuColumns { get; set; } = new();
    [Parameter] public int Columns { get; set; } = 4;
    [Parameter] public bool IsOpen { get; set; }
    [Parameter] public EventCallback<MegaMenuItem> OnItemClick { get; set; }

    private void Toggle()
    {
        IsOpen = !IsOpen;
    }

    private async Task HandleItemClick(MegaMenuItem item)
    {
        await OnItemClick.InvokeAsync(item);
        IsOpen = false;
    }
}
